// ═══════════════════════════════════════════════════════════════ // STB MULTI-STRATEGY ENGINE v3 // Strategies: SMC · Trend Following · Breakout · Mean Reversion · Gann // SL Types: Structural · S/R · ATR · OrderBlock · Candle-Based // ═══════════════════════════════════════════════════════════════ // ── AUTO TRADE SETTINGS ── // ── AUTO TRADE CONFIG ── // ── DEMO STATE ── // ══════════════════════════════════════════════════════════════ // SL CALCULATION ENGINE — 5 روش مختلف // ══════════════════════════════════════════════════════════════ function calcSL(dir, price, candles5m, candles4h, sw4h, sw15m, atr, nearOB, slTypeOverride) { const slType = slTypeOverride || AT_SETTINGS?.slType || 'atr'; // ── حداقل و حداکثر فاصله SL ── // برای کریپتو: حداقل 0.4%، حداکثر 4% // برای فارکس: حداقل 0.15%، حداکثر 2% const isCrypto = price > 0.01; const minPct = isCrypto ? 0.004 : 0.0015; const maxPct = isCrypto ? 0.04 : 0.02; const minDist = price * minPct; const maxDist = price * maxPct; // اطمینان از ATR معقول // برای قیمت‌های خیلی کوچیک (FLOKI, SHIB, PEPE...) const minPriceUnit = price < 0.001 ? price * 0.05 : price < 0.01 ? price * 0.03 : price * 0.003; const safeATR = Math.max(atr || minPriceUnit * 1.5, minPriceUnit); function validate(sl) { if (!sl || isNaN(sl)) return dir==='buy' ? price - safeATR*1.5 : price + safeATR*1.5; const dist = Math.abs(price - sl); // خیلی نزدیک → دور کن if (dist < minDist) { return dir==='buy' ? price - Math.max(safeATR*1.5, minDist*1.2) : price + Math.max(safeATR*1.5, minDist*1.2); } // خیلی دور → نزدیک کن if (dist > maxDist) { return dir==='buy' ? price - maxDist : price + maxDist; } // SL باید آن طرف قیمت باشه if (dir==='buy' && sl >= price) return price - Math.max(safeATR*1.5, minDist*1.2); if (dir==='sell' && sl <= price) return price + Math.max(safeATR*1.5, minDist*1.2); return sl; } // ── 1. STRUCTURAL: آخرین Low/High ── function structuralSL() { if (dir === 'buy') { const lows = [...(sw15m?.l||[]), ...(sw4h?.l||[])] .filter(l => l.p < price * 0.998) .sort((a,b) => b.p - a.p); if (lows[0]) return lows[0].p * 0.9985; } else { const highs = [...(sw15m?.h||[]), ...(sw4h?.h||[])] .filter(h => h.p > price * 1.002) .sort((a,b) => a.p - b.p); if (highs[0]) return highs[0].p * 1.0015; } return atrSL(); } // ── 2. S/R: حمایت/مقاومت ── function srSL() { const step = price > 50000?1000 : price > 10000?500 : price > 1000?50 : price > 100?5 : price > 1?0.5 : 0.001; const levels = []; [...(sw4h?.h||[]), ...(sw4h?.l||[]), ...(sw15m?.h||[]), ...(sw15m?.l||[])] .forEach(s => levels.push(s.p)); levels.push(Math.floor(price/step)*step, Math.ceil(price/step)*step); if (dir === 'buy') { const sup = levels.filter(l => l < price*0.998).sort((a,b)=>b-a); if (sup[0]) return sup[0] * 0.999; } else { const res = levels.filter(l => l > price*1.002).sort((a,b)=>a-b); if (res[0]) return res[0] * 1.001; } return structuralSL(); } // ── 3. ATR ── function atrSL() { const mult = Math.max(AT_SETTINGS?.atrMultSL||1.5, 1.2); return dir==='buy' ? price - safeATR*mult : price + safeATR*mult; } // ── 4. ORDER BLOCK ── function obSL() { if (dir==='buy' && nearOB?.ob_d) { const obLow = nearOB.ob_d.lo || nearOB.ob_d.price; if (obLow && obLow < price * 0.998) return obLow * 0.998; } if (dir==='sell' && nearOB?.ob_s) { const obHigh = nearOB.ob_s.hi || nearOB.ob_s.price; if (obHigh && obHigh > price * 1.002) return obHigh * 1.002; } return atrSL(); } // ── 5. CANDLE-BASED ── function candleSL() { const n = Math.max(AT_SETTINGS?.candlesSL||3, 2); const recent = (candles5m||[]).slice(-n-1, -1); if (!recent.length) return atrSL(); return dir==='buy' ? Math.min(...recent.map(c=>c.l)) * 0.9985 : Math.max(...recent.map(c=>c.h)) * 1.0015; } // ── 6. GANN ── function gannSL() { const swH = (sw4h?.h||[]).filter(h=>h.p>price).sort((a,b)=>a.p-b.p)[0]?.p || price*1.05; const swL = (sw4h?.l||[]).filter(l=>l.pb.p-a.p)[0]?.p || price*0.95; const gann382 = swL + (swH-swL)*0.382; const gann618 = swL + (swH-swL)*0.618; return dir==='buy' ? gann382*0.999 : gann618*1.001; } // ── AUTO: بهترین روش بر اساس شرایط ── function autoSL() { if (nearOB?.ob_d && dir==='buy') return obSL(); if (nearOB?.ob_s && dir==='sell') return obSL(); const strSL = structuralSL(); if (strSL) return strSL; return atrSL(); } const map = { structural: structuralSL, sr: srSL, atr: atrSL, orderblock: obSL, candle: candleSL, gann: gannSL, auto: autoSL, }; const rawSL = (map[slType] || autoSL)(); return validate(rawSL); } // ══════════════════════════════════════════════════════════════ // AUTO-SELECT STRATEGY based on market condition // ══════════════════════════════════════════════════════════════ // ══════════════════════════════════════════════════════════════ // SIGNAL ENGINE v3 — Multi-Strategy // ══════════════════════════════════════════════════════════════ // ══════════════════════════════════════════════════════════════ // AUTO TRADE TRIGGER // ══════════════════════════════════════════════════════════════
