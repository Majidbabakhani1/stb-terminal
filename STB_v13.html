STRUCTURAL: آخرین Low/High معتبر ── function structuralSL() { if (dir === 'buy') { // آخرین Low معتبر زیر قیمت const lows = [...(sw15m?.l||[]), ...(sw4h?.l||[])].filter(l => l.p < price).sort((a,b)=>b.p-a.p); const lastLow = lows[0]?.p; if (lastLow && Math.abs(price-lastLow)/price < 0.05) return lastLow * 0.9985; // کمی پایین‌تر return price - atr * AT_SETTINGS.atrMultSL; } else { const highs = [...(sw15m?.h||[]), ...(sw4h?.h||[])].filter(h => h.p > price).sort((a,b)=>a.p-b.p); const lastHigh = highs[0]?.p; if (lastHigh && Math.abs(price-lastHigh)/price < 0.05) return lastHigh * 1.0015; return price + atr * AT_SETTINGS.atrMultSL; } } // ── 2. S/R: نزدیک‌ترین سطح حمایت/مقاومت ── function srSL() { // ترکیب swing + round numbers const allLevels = []; [...(sw4h?.h||[]), ...(sw4h?.l||[]), ...(sw15m?.h||[]), ...(sw15m?.l||[])].forEach(s => { allLevels.push(s.p); }); // Round numbers (مضارب ۱۰۰ برای BTC، ۰.۰۰۱ برای فارکس) const step = price > 10000 ? 500 : price > 1000 ? 50 : price > 100 ? 5 : price > 1 ? 0.5 : 0.001; const roundLow = Math.floor(price/step)*step; const roundHigh = Math.ceil(price/step)*step; allLevels.push(roundLow, roundHigh); if (dir === 'buy') { const supports = allLevels.filter(l => l < price * 0.999).sort((a,b)=>b-a); if (supports[0] && Math.abs(price-supports[0])/price < 0.04) return supports[0] * 0.999; } else { const resistances = allLevels.filter(l => l > price * 1.001).sort((a,b)=>a-b); if (resistances[0] && Math.abs(price-resistances[0])/price < 0.04) return resistances[0] * 1.001; } return structuralSL(); } // ── 3. ATR-BASED ── function atrSL() { const mult = AT_SETTINGS.atrMultSL; return dir === 'buy' ? price - atr * mult : price + atr * mult; } // ── 4. ORDER BLOCK ── function obSL() { if (dir === 'buy' && nearOB?.ob_d) { return nearOB.ob_d.lo * 0.9985; // پایین‌تر از Demand OB } if (dir === 'sell' && nearOB?.ob_s) { return nearOB.ob_s.hi * 1.0015; // بالاتر از Supply OB } return atrSL(); } // ── 5. GANN 50% LEVEL ── function gannSL() { const swH = (sw4h?.h||[]).filter(h=>h.p>price).sort((a,b)=>a.p-b.p)[0]?.p || price*1.03; const swL = (sw4h?.l||[]).filter(l=>l.pb.p-a.p)[0]?.p || price*0.97; const gann50 = (swH + swL) / 2; const gann382 = swL + (swH-swL)*0.382; const gann618 = swL + (swH-swL)*0.618; if (dir === 'buy') return Math.max(gann382, price-atr*2) * 0.999; else return Math.min(gann618, price+atr*2) * 1.001; } // ── AUTO SELECT ── if (slType === 'auto') { if (nearOB?.ob_d || nearOB?.ob_s) return obSL(); const trend4h = candles4h?.length >= 21; if (trend4h) return structuralSL(); return atrSL(); } const map = {structural:structuralSL, sr:srSL, atr:atrSL, orderblock:obSL, candle:candleSL, gann:gannSL}; return (map[slType] || atrSL)(); } // ══════════════════════════════════════════════════════════════ // AUTO-SELECT STRATEGY based on market condition // ══════════════════════════════════════════════════════════════ function autoSelectStrategy(trendH4, rsi5m, volSurge, nearOB) { // بازار رنج if (trendH4 === 'side') { if (nearOB) return 'SMC'; return 'MEAN_REV'; } // بازار ترند قوی if (trendH4 !== 'side' && volSurge) { return 'TREND'; } // شکست سطح if (nearOB && volSurge) return 'BREAKOUT'; // پیش‌فرض return 'SMC'; } // ══════════════════════════════════════════════════════════════ // SIGNAL ENGINE v3 — Multi-Strategy // ══════════════════════════════════════════════════════════════ // ══════════════════════════════════════════════════════════════ // AUTO TRADE TRIGGER // ══════════════════════════════════════════════════════════════
